// --- ACTUALIZACIÓN ROBUSTA DESDE GITHUB GIST --- async function tryUpdate() { const badge = document.getElementById('status-badge'); badge.innerText = "ACTUALIZANDO..."; // Si en 5 segundos no responde, volvemos a LOCAL const timeoutId = setTimeout(() => { badge.innerText = "LOCAL"; console.log("Timeout: Usando datos locales."); }, 5000); // ---------------------------------------------------------- // IMPORTANTE: PEGA AQUÍ TU URL RAW (Botón Raw de tu Gist) // ---------------------------------------------------------- const GIST_RAW_URL = 'https://gist.githubusercontent.com/RenzoCL/7f56fdb7784efa4aa8e135f0968a0dc1/raw/5ad8f9d6fb1aa48257f0dd342f9f3da50bcbb7f2/eventos.json'; try { // Usamos un proxy para evitar bloqueos del navegador const proxyUrl = 'https://corsproxy.io/?'; const response = await fetch(proxyUrl + encodeURIComponent(GIST_RAW_URL)); // Los datos ya vienen ordenados y agrupados desde tu script Python const data = await response.json(); clearTimeout(timeoutId); if (data && data.length > 0) { currentList = data; // Cargamos directo renderAgenda(); badge.innerText = "EN VIVO"; badge.style.color = "#4ecca3"; // Verde console.log(`✅ Datos actualizados: ${data.length} eventos.`); } else { badge.innerText = "SIN DATOS"; } } catch (e) { clearTimeout(timeoutId); console.error("Error actualizando:", e); badge.innerText = "ERROR RED"; } }
